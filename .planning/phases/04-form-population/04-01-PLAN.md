               context['youtube_import'] = youtube_import
               context['has_youtube_import'] = True

    4. Add import processing after successful form save:
       In form_valid(), after recipe is saved:
       ```python
       def form_valid(self, form):
           youtube_import = self.request.session.get('youtube_import')
           
           # Save recipe first
           recipe = form.save(commit=False)
           recipe.household = self.request.user.household
           recipe.save()
           
           # Process ingredients from YouTube import
           if youtube_import and form.cleaned_data.get('use_youtube_data', False):
               ingredients = youtube_import.get('ingredients', [])
               for ing in ingredients:
                   IngredientLink.objects.create(
                       recipe=recipe,
                       quantity=ing.get('quantity', ''),
                       unit=ing.get('unit', ''),
                       notes=ing.get('notes', ''),
                   )
               
               # Process instructions
               instructions = youtube_import.get('instructions', [])
               for inst in instructions:
                   Instruction.objects.create(
                       recipe=recipe,
                       step_number=inst.get('step_number', 1),
                       text=inst.get('text', ''),
                   )
               
               # Clear session after successful import
               del self.request.session['youtube_import']
               
           return redirect(recipe.get_absolute_url())
       ```

           """Extract ingredients from form data."""
           ingredients = []
           for key, value in self.data.items():
               if key.startswith('ingredient_name_'):
                   idx = key.split('_')[-1]
                   ingredients.append({
                       'name': value,
                       'quantity': self.data.get(f'ingredient_quantity_{idx}', ''),
