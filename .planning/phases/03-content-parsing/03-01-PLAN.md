    2. Modify form_valid method to call parser after fetching metadata:
       ```python
       def form_valid(self, form):
           youtube_url = form.cleaned_data['youtube_url']
           
           # Get API key from settings
           api_key = getattr(settings, 'YOUTUBE_API_KEY', None)
           if not api_key:
               form.add_error('youtube_url', 'YouTube API key not configured.')
               return self.form_invalid(form)
           
           try:
               # Parse video ID and fetch metadata (existing Phase 2 code)
               youtube_service = YouTubeService(api_key)
               video_id = youtube_service.extract_video_id(youtube_url)
               metadata = youtube_service.get_video_metadata(video_id)
               
               # NEW: Parse description into ingredients and instructions
               parser = RecipeParsingService()
               ingredients = parser.parse_ingredients(metadata.description)
               instructions = parser.parse_instructions(metadata.description)
               unparsed = parser.identify_unparseable(metadata.description.split('\n'))
               
               # Store in session for Phase 4 pre-population
               self.request.session['youtube_import'] = {
                   'video_id': metadata.video_id,
                   'title': metadata.title,
                   'description': metadata.description,
                   'thumbnail_url': metadata.thumbnail_url,
                   'ingredients': [
                       {
                           'name': i.name,
                           'quantity': i.quantity,
                           'unit': i.unit,
                           'notes': i.notes,
                       }
                       for i in ingredients
                   ],
                   'instructions': [
                       {
                           'step_number': i.step_number,
                           'text': i.text,
                           'timestamp': i.timestamp,
                       }
                       for i in instructions
                   ],
                   'unparsed_lines': unparsed,
               }
               
               # Redirect to recipe create (Phase 4 handles pre-population)
               return redirect('recipe_create')
               
           except InvalidVideoError as e:
               form.add_error('youtube_url', str(e))
               return self.form_invalid(form)
           except APIError as e:
               form.add_error('youtube_url', str(e))
               return self.form_invalid(form)
           except Exception as e:
               form.add_error('youtube_url', 'Could not fetch or parse video. Please try again.')
               return self.form_invalid(form)
       ```
