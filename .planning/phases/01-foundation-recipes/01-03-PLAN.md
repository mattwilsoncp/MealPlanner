  <files>ingredients/models.py</files>

  <action>
    1. Create ingredients/models.py:
       - from django.db import models
       - from household.models import Household
       
       - class Ingredient(models.Model):
         - name = models.CharField(max_length=100, db_index=True)
         - household = models.ForeignKey(Household, on_delete=models.CASCADE)
         - usda_food_id = models.CharField(max_length=20, blank=True)  # USDA FoodData Central ID
         - created_at = models.DateTimeField(auto_now_add=True)
         
         - class Meta:
           - unique_together = ['household', 'name']
           - ordering = ['name']
         
         - def __str__(self): return self.name
       
       - class IngredientLink(models.Model):
         - UNIT_CHOICES = [
             - ('oz', 'ounce'),
             - ('lb', 'pound'),
             - ('cup', 'cup'),
             - ('tbsp', 'tablespoon'),
             - ('tsp', 'teaspoon'),
             - ('g', 'gram'),
             - ('kg', 'kilogram'),
             - ('ml', 'milliliter'),
             - ('l', 'liter'),
             - ('piece', 'piece'),
             - ('clove', 'clove'),
             - ('slice', 'slice'),
             - ('bunch', 'bunch'),
             - ('can', 'can'),
           ]
         - recipe = models.ForeignKey('recipes.Recipe', on_delete=models.CASCADE)
         - ingredient = models.ForeignKey(Ingredient, on_delete=models.CASCADE)
         - quantity = models.DecimalField(max_digits=6, decimal_places=2)
         - unit = models.CharField(max_length=20, choices=UNIT_CHOICES)
         - order = models.PositiveIntegerField(default=0)
         - inventory_item = models.ForeignKey('inventory.InventoryItem', null=True, blank=True, on_delete=models.SET_NULL)
         
         - class Meta:
           - ordering = ['order']
         
         - def __str__(self): return f"{self.quantity} {self.unit} {self.ingredient.name}"
  </action>
  <acceptance_criteria>
    - ingredients/models.py contains class Ingredient and IngredientLink
    - Ingredient has name, household, usda_food_id
    - IngredientLink has recipe, ingredient, quantity, unit, order, inventory_item
    - Ingredient has db_index on name
    - IngredientLink has UNIT_CHOICES
  </acceptance_criteria>
  <verify>

  </verify>

  <files>ratings/models.py</files>
  <action>
    1. Create ratings/models.py:
       - from django.db import models
       - from django.contrib.auth import get_user_model
       
       - class Rating(models.Model):
         - RATING_CHOICES = [(i, str(i)) for i in range(1, 6)]
         - recipe = models.ForeignKey('recipes.Recipe', on_delete=models.CASCADE)
         - user = models.ForeignKey(get_user_model(), on_delete=models.CASCADE)
         - score = models.IntegerField(choices=RATING_CHOICES)
         - notes = models.TextField(blank=True)
         - created_at = models.DateTimeField(auto_now_add=True)
         - updated_at = models.DateTimeField(auto_now=True)
         
         - class Meta:
           - unique_together = ['recipe', 'user']
           - ordering = ['-created_at']
         
         - def __str__(self): return f"{self.recipe.title}: {self.score}/5"
  </action>
  <acceptance_criteria>
